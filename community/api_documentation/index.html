Quick Start Guide

1

Installation

Set up your MultiOS development environment in minutes.

# Install MultiOS development tools
curl -fsSL https://get.multios.dev | sh

# Set up development environment
multios setup --complete
2

Your First Program

Create and run your first MultiOS application.

use multios_kernel::process::{create_process, ProcessCreateParams};

fn main() -> Result<(), Box> {
    let params = ProcessCreateParams {
        name: "hello_multios".to_string(),
        command_line: vec!["hello_multios".to_string()],
        priority: ProcessPriority::Normal,
        ..Default::default()
    };
    
    let process_id = create_process(params)?;
    println!("Created process: {}", process_id);
    
    Ok(())
}
