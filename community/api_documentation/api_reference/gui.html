Quick Start

Create Your First Window

use multi::gui::{Application, Window, Button, Label};

// Create application instance
let mut app = Application::new("My First MultiOS App")?;

// Create main window
let mut window = Window::builder()
    .title("Hello MultiOS")
    .size(800, 600)
    .centered(true)
    .build()?;

// Create a label
let mut label = Label::builder()
    .text("Welcome to MultiOS GUI!")
    .font_size(24)
    .text_color(Color::from_rgb(0, 100, 200))
    .position(50, 50)
    .build()?;

// Create a button
let mut button = Button::builder()
    .text("Click Me!")
    .size(150, 50)
    .position(50, 150)
    .build()?;

// Add event handler
button.on_click(Box::new(|| {
    println!("Button clicked!");
}));

// Add widgets to window
window.add_widget(&mut label);
window.add_widget(&mut button);

// Show window
window.show()?;

// Run application
app.run()?;

Application Framework

MultiOS provides a comprehensive application framework for GUI development.

Application

Main application container that manages windows and event processing.

Create and Run Application

use multi::gui::{Application, ApplicationConfig, EventLoop};

// Application configuration
let config = ApplicationConfig {
    title: "MultiOS Desktop App",
    window_count: 5,
    enable_menus: true,
    enable_keyboard_shortcuts: true,
};

// Create application with custom config
let mut app = Application::with_config(config)?;

// Set application-wide event handlers
app.on_menu_event(Box::new(|event| {
    match event {
        MenuEvent::Exit => app.quit(),
        MenuEvent::About => show_about_dialog(),
        _ => {}
    }
}));

app.on_window_close(Box::new(|window| {
    println!("Window {} closed", window.title());
    true // Allow window to close
}));

// Run the application
let exit_code = app.run_event_loop()?;

Event System

Comprehensive event handling for interactive user interfaces.

Event Handling

Handle Widget Events

use multi::gui::{WidgetEvent, EventHandler, MouseEvent, KeyEvent};

// Button click handling
button.on_click(Box::new(|| {
    println!("Button was clicked!");
    // Update UI, process data, etc.
}));

// Text input changes
text_input.on_text_changed(Box::new(|text| {
    println!("Text changed: {}", text);
    // Validate input, auto-save, etc.
}));

// Mouse events
canvas.on_mouse_down(Box::new(|event| {
    match event {
        MouseEvent::LeftDown(pos) => {
            println!("Mouse down at {:?}", pos);
            start_drawing(pos);
        }
        MouseEvent::RightDown(pos) => {
            show_context_menu(pos);
        }
        _ => {}
    }
}));

canvas.on_mouse_move(Box::new(|pos| {
    if is_drawing() {
        continue_drawing(pos);
    }
}));

// Keyboard events
window.on_key_down(Box::new(|event| {
    match event.key {
        Key::Escape => window.close(),
        Key::F1 => show_help_dialog(),
        Key::Enter if event.ctrl_key => save_document(),
        _ => {}
    }
}));

// Focus events
text_input.on_focus_gained(Box::new(|| {
    text_input.set_background_color(Color::LightBlue);
}));

text_input.on_focus_lost(Box::new(|| {
    text_input.set_background_color(Color::White);
}));

Widget Animations

use multi::gui::{Animation, Easing, AnimationConfig};

// Create fade-in animation
let fade_in = Animation::new()
    .property("opacity")
    .from(0.0)
    .to(1.0)
    .duration(Duration::from_millis(500))
    .easing(Easing::CubicOut)

    .property("position")

    .duration(Duration::from_millis(800))
    .easing(Easing::ElasticOut);

    .property("scale")
    .from(0.5)
    .to(1.0)

Responsive Design

Build adaptive interfaces that work across different screen sizes and resolutions.

Flexible Layouts

Create Responsive Layouts

use multi::gui::{ResponsiveLayout, Breakpoint, SizeClass};

// Responsive layout with breakpoints
let mut layout = ResponsiveLayout::builder()
    .breakpoints(vec![
        Breakpoint::new("mobile", 0, 600),
        Breakpoint::new("tablet", 600, 1024),
        Breakpoint::new("desktop", 1024, u32::MAX),
    ])
    .build()?;

// Configure layout for different screen sizes
layout.on_breakpoint("mobile", |layout| {
    layout.set_columns(1);
    layout.set_spacing(8);
    layout.set_padding(Padding::new(10, 10, 10, 10));
});

layout.on_breakpoint("tablet", |layout| {
    layout.set_columns(2);
    layout.set_spacing(12);
    layout.set_padding(Padding::new(15, 15, 15, 15));
});

layout.on_breakpoint("desktop", |layout| {
    layout.set_columns(3);
    layout.set_spacing(16);
    layout.set_padding(Padding::new(20, 20, 20, 20));
});

// Size-aware widgets
let mut responsive_button = Button::builder()
    .text("Responsive Button")
    .min_width(100)
    .max_width(300)
    .preferred_width(SizeClass::Flexible)
    .size_mode(SizeMode::Fill)
    .build()?;

}));

Error Handling

Comprehensive error handling for GUI operations.

GUI Error Types

Handle GUI Errors

use multi::gui::{GuiError, ErrorKind};

fn create_robust_ui() -> Result<Window, String> {
    match Window::builder()
        .title("Robust UI")
        .size(800, 600)
        .build()
    {
        Ok(mut window) => {
            // Handle widget creation errors
            match Label::builder().text("Test").build() {
                Ok(mut label) => {
                    match window.add_widget(&mut label) {
                        Ok(_) => println!("Widget added successfully"),
                        Err(e) => {
                            match e.error_type {
                                ErrorKind::WidgetCapacityExceeded => {
                                    return Err("Too many widgets in window".to_string());
                                }
                                ErrorKind::InvalidWidget => {
                                    return Err("Invalid widget configuration".to_string());
                                }
                                _ => return Err(format!("Widget error: {}", e))
                            }
                        }
                    }
                }
                Err(e) => return Err(format!("Label creation failed: {}", e))
            }
            
            // Handle window operation errors
            match window.show() {
                Ok(_) => Ok(window),
                Err(e) => match e.error_type {
                    ErrorKind::DisplayError => Err("Display connection failed".to_string()),
                    ErrorKind::PermissionDenied => Err("Window creation denied".to_string()),
                    _ => Err(format!("Window error: {}", e))
                }
            }
        }
        Err(e) => Err(format!("Failed to create window: {}", e))
    }
}
