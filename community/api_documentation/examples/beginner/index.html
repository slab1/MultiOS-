Basic Memory Allocation

Learn how to allocate and manage memory in MultiOS using the memory management API.

Beginner 15 min
Rust 
use multios_memory::alloc::{allocate, deallocate, MemoryRegion};

// Allocate a new memory region
fn allocate_example() -> Result<MemoryRegion, Box<dyn std::error::Error>> {
    let region = allocate(1024, MemoryAlignment::PageAligned)?;
    
    // Write some data to the allocated memory
    let data = "Hello, MultiOS!";
    region.write(0, data.as_bytes())?;
    
    println!("Allocated memory region: {:?}", region);
    println!("Data written: {}", String::from_utf8_lossy(®ion.read(0, data.len())?));
    
    Ok(region)
}

fn main() -> Result<(), Box<dyn std::error::Error>> {
    let region = allocate_example()?;
    
    // Clean up
    deallocate(region)?;
    println!("Memory deallocated successfully");
    
    Ok(())
}

Create Your First Process

Learn how to create, manage, and terminate processes in MultiOS.

Beginner 20 min
Rust 
use multios_process::{ProcessManager, ProcessId, ProcessCreateParams, ProcessPriority};

fn main() -> Result<(), Box<dyn std::error::Error>> {
    let mut process_manager = ProcessManager::new();
    
    // Create a simple process
    let create_params = ProcessCreateParams {
        name: "hello_world".to_string(),
        command_line: vec!["hello_world".to_string()],
        priority: ProcessPriority::Normal,
        stack_size: 4096,
        heap_size: 1024,
        ..Default::default()
    };
    
    let process_id = process_manager.create_process(create_params)?;
    println!("Created process with ID: {:?}", process_id);
    
    // List all processes
    let processes = process_manager.list_processes()?;
    for process in processes {
        println!("Process: {} - {}", process.id, process.name);
    }
    
    // Wait for the process to complete
    let exit_code = process_manager.wait_for_process(process_id)?;
    println!("Process exited with code: {}", exit_code);
    
    Ok(())
}
Output
Created process with ID: ProcessId(123)
Process: ProcessId(123) - hello_world
Process: ProcessId(456) - init
Process exited with code: 0

File Operations

Learn basic file operations including creating, reading, writing, and deleting files.

Beginner 25 min
Rust 
use multios_filesystem::{FileManager, FileOpenMode, Path};

fn main() -> Result<(), Box<dyn std::error::Error>> {
    let mut file_manager = FileManager::new();
    
    let file_path = Path::new("/tmp/hello_multios.txt");
    
    // Create and write to a file
    {
        let mut file = file_manager.create_file(
            file_path,
            FileOpenMode::Write | FileOpenMode::Create
        )?;
        
        let content = "Hello from MultiOS filesystem!\nThis is a test file.";
        file.write_all(content.as_bytes())?;
        println!("File written: {}", file_path);
    }
    
    // Read the file back
    {
        let mut file = file_manager.open_file(file_path, FileOpenMode::Read)?;
        
        let mut content = String::new();
        file.read_to_string(&mut content)?;
        
        println!("File content:\n{}", content);
    }
    
    // Check file exists
    if file_manager.exists(file_path)? {
        let metadata = file_manager.metadata(file_path)?;
        println!("File size: {} bytes", metadata.size);
    }
    
    // Clean up
    file_manager.delete_file(file_path)?;
    println!("File deleted: {}", file_path);
    
    Ok(())
}
Output
File written: /tmp/hello_multios.txt
File content:
Hello from MultiOS filesystem!
This is a test file.
File size: 45 bytes
File deleted: /tmp/hello_multios.txt

System Information

Retrieve system information using the kernel API.

Beginner 10 min
Rust 
use multios_kernel::{SystemInfo, CPUInfo, KernelVersion};

fn main() -> Result<(), Box<dyn std::error::Error>> {
    // Get system information
    let system_info = SystemInfo::get()?;
    
    println!("MultiOS System Information");
    println!("=========================");
    println!("OS Name: {}", system_info.os_name);
    println!("Version: {}", system_info.version);
    println!("Architecture: {}", system_info.architecture);
    
    // Get CPU information
    let cpu_info = CPUInfo::get()?;
    println!("CPU Vendor: {}", cpu_info.vendor);
    println!("CPU Model: {}", cpu_info.model);
    println!("CPU Cores: {}", cpu_info.core_count);
    println!("CPU Features: {:?}", cpu_info.features);
    
    // Get memory information
    println!("Total Memory: {} bytes", system_info.total_memory);
    println!("Available Memory: {} bytes", system_info.available_memory);
    
    // Get kernel version
    let kernel_version = KernelVersion::get()?;
    println!("Kernel Version: {}", kernel_version);
    
    Ok(())
}

Network Configuration

Configure and query network interfaces using MultiOS networking APIs.

Beginner 20 min
Rust 
use multios_network::{NetworkManager, NetworkInterface, IPAddress};

fn main() -> Result<(), Box<dyn std::error::Error>> {
    let network_manager = NetworkManager::new();
    
    // List all network interfaces
    let interfaces = network_manager.list_interfaces()?;
    
    println!("Network Interfaces:");
    for interface in &interfaces {
        println!("  Name: {}", interface.name);
        println!("  Type: {}", interface.interface_type);
        println!("  MAC Address: {}", interface.mac_address);
        
        if let Some(ip_address) = interface.ip_address {
            println!("  IP Address: {}", ip_address);
        }
        
        if let Some(netmask) = interface.netmask {
            println!("  Netmask: {}", netmask);
        }
        
        println!("  Status: {}", if interface.is_up { "Up" } else { "Down" });
        println!();
    }
    
    // Configure a static IP (if interface exists)
    if let Some(eth0) = interfaces.iter().find(|i| i.name == "eth0") {
        if !eth0.is_up {
            network_manager.set_interface_up(ð0.name)?;
            println!("Interface {} is now up", eth0.name);
        }
        
        // Set IP address
        let new_ip = IPAddress::from_str("192.168.1.100")?;
        network_manager.set_ip_address(ð0.name, new_ip)?;
        println!("Set IP address for {}: {}", eth0.name, new_ip);
    }
    
    Ok(())
}

Create a Window

Create your first GUI window using the MultiOS window manager.

Beginner 30 min
Rust 
use multios_gui::{
    WindowManager, Window, WindowConfig, WindowFlags,
    Color, Point, Size
};

fn main() -> Result<(), Box<dyn std::error::Error>> {
    let mut window_manager = WindowManager::new()?;
    
    // Create a new window
    let window_config = WindowConfig {
        title: "My First MultiOS Window".to_string(),
        position: Point { x: 100, y: 100 },
        size: Size { width: 800, height: 600 },
        flags: WindowFlags::RESIZEABLE | WindowFlags::DECORATED,
        background_color: Color::from_rgb(240, 248, 255),
        ..Default::default()
    };
    
    let mut window = window_manager.create_window(window_config)?;
    
    // Add some content
    window.draw_text(
        Point { x: 50, y: 50 },
        "Hello from MultiOS GUI!",
        Color::from_rgb(0, 0, 0),
        24
    )?;
    
    // Add a button
    window.draw_button(
        Point { x: 50, y: 100 },
        Size { width: 200, height: 40 },
        "Click Me!",
        Color::from_rgb(0, 123, 255),
        Color::from_rgb(255, 255, 255)
    )?;
    
    // Show the window
    window.show()?;
    window_manager.set_active_window(window.id())?;
    
    // Handle window events (simplified)
    for event in window_manager.poll_events()? {
        match event {
            GUIEvent::WindowClosed(_) => {
                println!("Window closed");
                break;
            }
            GUIEvent::ButtonClicked(id) => {
                println!("Button clicked: {:?}", id);
                window.set_title("Button was clicked!")?;
            }
            _ => {}
        }
    }
    
    Ok(())
}
