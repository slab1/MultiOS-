from multios import FileSystem, Path
import json

def file_system_demo():
    """Demonstrate file system operations"""
    fs = FileSystem()
    
    # Working directory
    current_dir = fs.get_current_dir()
    print(f"Current directory: {current_dir}")
    
    # Create directories
    app_dir = Path("/apps/my_app")
    fs.create_directory(app_dir, recursive=True)
    
    # Write data
    config = {
        "app_name": "My App",
        "version": "1.0.0",
        "settings": {
            "theme": "dark",
            "language": "en"
        }
    }
    
    config_path = app_dir / "config.json"
    with fs.open(config_path, "w") as f:
        json.dump(config, f, indent=2)
    
    # Read data
    with fs.open(config_path, "r") as f:
        loaded_config = json.load(f)
    
    print(f"Loaded config: {loaded_config}")
    
    # List directory
    files = fs.list_directory(app_dir)
    for file in files:
        print(f"File: {file.name} ({file.size} bytes)")
    
    # File operations
    fs.copy(config_path, app_dir / "config_backup.json")
    fs.move(app_dir / "config_backup.json", app_dir / "old_config.json")
    
    return app_dir

def directory_traversal():
    """Advanced directory operations"""

Process Management

Create and manage processes from Python:

Process Management 
import time
import threading
from multios import Process, ProcessManager

class WorkerProcess:
    def __init__(self):
        self.pm = ProcessManager()
        self.running = False
    
    def start_worker(self, worker_id):
        """Start a worker process"""
        def worker():
            worker_process = Process.get_current()
            worker_process.set_name(f"Worker-{worker_id}")
            
            print(f"Worker {worker_id} started with PID: {worker_process.pid}")
            
            self.running = True
            counter = 0
            while self.running:
                print(f"Worker {worker_id}: {counter}")
                counter += 1
                time.sleep(1)
            
            print(f"Worker {worker_id} stopped")
        
        # Start worker in background
        thread = threading.Thread(target=worker)
        thread.daemon = True
        thread.start()
        
        return thread
    
    def spawn_subprocess(self, command):
        """Spawn a subprocess"""
        try:
            process = self.pm.spawn(command)
            print(f"Subprocess started: {process.pid}")
            return process
        except Exception as e:
            print(f"Failed to spawn subprocess: {e}")
            return None
    
    def monitor_processes(self):
        """Monitor all running processes"""
        processes = self.pm.list_processes()
        
        print("Running processes:")
        for proc in processes:
            print(f"  PID {proc.pid}: {proc.name} - {proc.status}")
    
    def run_example(self):
        """Run process management example"""
        # Start multiple workers
        workers = []
        for i in range(3):
            worker = self.start_worker(i + 1)
            workers.append(worker)
        
        # Run for a few seconds
        time.sleep(5)
        
        # Stop workers
        self.running = False
        
        # Spawn subprocess
        subprocess = self.spawn_subprocess("echo 'Hello from subprocess'")
        time.sleep(1)
        
        # Monitor processes
        self.monitor_processes()

# Usage
if __name__ == "__main__":
    worker_manager = WorkerProcess()
    worker_manager.run_example()

Network Operations

Network programming with Python on MultiOS:

Network Server 
import json
import threading
import time
from multios import Network, Socket, TCPServer

class SimpleHTTPServer:
    def __init__(self, host="127.0.0.1", port=8080):
        self.host = host
        self.port = port
        self.running = False
    
    def handle_request(self, request):
        """Handle incoming HTTP request"""
        method = request.method
        path = request.path
        
        # Simple routing
        if path == "/":
            return self.json_response(200, {
                "message": "Hello from MultiOS Python!",
                "method": method,
                "timestamp": time.time()
            })
        elif path == "/status":
            return self.json_response(200, {
                "status": "running",
                "uptime": time.time() - self.start_time
            })
        else:
            return self.json_response(404, {"error": "Not found"})
    
    def json_response(self, status_code, data):
        """Create JSON response"""
        response = {
            "status_code": status_code,
            "headers": {"Content-Type": "application/json"},
            "body": json.dumps(data, indent=2)
        }
        return response
    
    def start_server(self):
        """Start the HTTP server"""
        self.running = True
        self.start_time = time.time()
        
        def handle_client(client_socket, address):
            """Handle individual client connection"""
            try:
                # Receive request
                request_data = client_socket.recv(1024)
                if not request_data:
                    return
                
                # Parse request (simplified)
                request_lines = request_data.decode().split('\r\n')
                request_line = request_lines[0]
                parts = request_line.split(' ')
                
                if len(parts) >= 2:
                    method = parts[0]
                    path = parts[1]
                    
                    # Create request object
                    class Request:
                        def __init__(self, method, path):
                            self.method = method
                            self.path = path
                    
                    request = Request(method, path)
                    
                    # Handle request
                    response = self.handle_request(request)
                    
                    # Send response
                    response_data = f"HTTP/1.1 {response['status_code']} OK\r\n"
                    response_data += f"Content-Type: {response['headers']['Content-Type']}\r\n"
                    response_data += f"Content-Length: {len(response['body'])}\r\n"
                    response_data += "\r\n"
                    response_data += response['body']
                    
                    client_socket.send(response_data.encode())
            except Exception as e:
                print(f"Error handling client: {e}")
            finally:
                client_socket.close()
        
        # Create server socket
        server_socket = Socket()
        server_socket.bind((self.host, self.port))
        server_socket.listen(5)
        
        print(f"Server started on {self.host}:{self.port}")
        
        try:
            while self.running:
                # Accept connection
                client_socket, address = server_socket.accept()
                
                # Handle in separate thread
                client_thread = threading.Thread(
                    target=handle_client,
                    args=(client_socket, address)
                )
                client_thread.daemon = True
                client_thread.start()
        except KeyboardInterrupt:
            print("Server shutting down...")
        finally:
            server_socket.close()
    
    def stop_server(self):
        """Stop the server"""
        self.running = False

# Usage
if __name__ == "__main__":
    server = SimpleHTTPServer()
    
    # Start server in background
    server_thread = threading.Thread(target=server.start_server)
    server_thread.daemon = True
    server_thread.start()
    
    try:
        # Keep main thread alive
        while True:
            time.sleep(1)
    except KeyboardInterrupt:
        server.stop_server()

Performance and Best Practices

Performance Considerations

  • Use native extensions: For performance-critical code, implement functions in Rust and call from Python
  • Avoid excessive object creation: Reuse objects and connections when possible
  • Batch operations: Group multiple operations to reduce system calls
  • Memory management: Use context managers and proper cleanup
Performance Example 
import time
import threading
from contextlib import contextmanager
from multios import FileSystem, Network

@contextmanager
def file_context(fs, path, mode="r"):
    """Context manager for file operations"""
    file_handle = None
    try:
        file_handle = fs.open(path, mode)
        yield file_handle
    finally:
        if file_handle:
            file_handle.close()

@contextmanager
def network_context(host, port):
    """Context manager for network operations"""
    socket = None
    try:
        socket = Network.Socket()
        socket.connect((host, port))
        yield socket
    finally:
        if socket:
            socket.close()

def batch_file_operations():
    """Perform batch file operations efficiently"""
    fs = FileSystem()
    directory = Path("/temp/batch_ops")
    
    # Create directory if it doesn't exist
    fs.create_directory(directory, recursive=True)
    
    # Batch write
    start_time = time.time()
    
    files_data = [
        f"data_{i:04d}.txt" for i in range(1000)
    ]
    
    # Write files in batch
    for i, filename in enumerate(files_data):
        file_path = directory / filename
        with file_context(fs, file_path, "w") as f:
            f.write(f"File {i} data\n" * 10)
    
    write_time = time.time() - start_time
    print(f"Wrote {len(files_data)} files in {write_time:.2f} seconds")
    
    # Batch read
    start_time = time.time()
    
    total_size = 0
    for filename in files_data:
        file_path = directory / filename
        with file_context(fs, file_path, "r") as f:
            content = f.read()
            total_size += len(content)
    
    read_time = time.time() - start_time
    print(f"Read {len(files_data)} files ({total_size} bytes) in {read_time:.2f} seconds")
    
    # Cleanup
    fs.remove_directory(directory, recursive=True)

def efficient_processing():
    """Demonstrate efficient processing patterns"""
    # Use threading for I/O operations
    def io_worker(data_queue, result_queue):
        while True:
            try:
                data = data_queue.get(timeout=1)
                if data is None:  # Poison pill to stop
                    break
                
                # Simulate I/O operation
                time.sleep(0.1)
                result = data * 2
                result_queue.put(result)
            except:
                break
    
    # Create worker threads
    data_queue = []
    result_queue = []
    
    threads = []
    for i in range(4):
        thread = threading.Thread(target=io_worker, args=(data_queue, result_queue))
        thread.start()
        threads.append(thread)
    
    # Process data
    for i in range(100):
        data_queue.append(i)
    
    # Collect results
    results = []
    for i in range(100):
        result = result_queue.get()
        results.append(result)
    
    # Stop workers
    for i in range(4):
        data_queue.append(None)
    
    for thread in threads:
        thread.join()
    
    print(f"Processed {len(results)} items")

Error Handling

Error Handling Pattern 
from multios import MultiOSError, FileSystemError, NetworkError

class AppError(Exception):
    """Base application error"""
    pass

class FileOperationError(AppError):
    """File operation error"""
    pass

class NetworkOperationError(AppError):
    """Network operation error"""
    pass

def safe_file_operation(fs, path, operation):
    """Perform file operation with proper error handling"""
    try:
        return operation()
    except FileSystemError as e:
        raise FileOperationError(f"File operation failed: {e}")
    except Exception as e:
        raise AppError(f"Unexpected error: {e}")

def safe_network_operation(operation):
    """Perform network operation with proper error handling"""
    try:
        return operation()
    except NetworkError as e:
        raise NetworkOperationError(f"Network operation failed: {e}")
    except Exception as e:
        raise AppError(f"Unexpected error: {e}")

# Usage
try:
    fs = FileSystem()
    
    def read_file():
        with fs.open("/my_app/config.txt", "r") as f:
            return f.read()
    
    content = safe_file_operation(fs, "/my_app/config.txt", read_file)
    print("File content:", content)
    
except FileOperationError as e:
    print(f"File error: {e}")
except AppError as e:
    print(f"App error: {e}")
except Exception as e:
    print(f"Unexpected error: {e}")

        try:
            # Your application code here
            self.perform_operations()
            
        except Exception as e:
            logger.exception(f"Application error: {e}")
            self.debug.print_stack_trace()
            raise
        finally:
            logger.info("Application ended")

        """Perform some operations"""
        logger.debug("Starting operations")
        
        # Debug information
        self.debug.log_memory_usage("After initialization")
        
        # Simulate some work
        import time
        for i in range(5):
            logger.debug(f"Operation {i+1}")
            time.sleep(0.1)
        
        logger.debug("Operations completed")
        self.debug.log_memory_usage("After operations")
    
    def profile_function(self, func, *args, **kwargs):
        """Profile a function call"""
        import time
        
        start_time = time.time()
        start_memory = self.debug.get_memory_usage()
        
        try:
            result = func(*args, **kwargs)
            return result
        finally:
            end_time = time.time()
            end_memory = self.debug.get_memory_usage()
            
            logger.info(f"Function {func.__name__} took {end_time - start_time:.3f}s")
            logger.info(f"Memory usage: {start_memory} -> {end_memory} bytes")
