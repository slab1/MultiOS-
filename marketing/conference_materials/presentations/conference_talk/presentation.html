// Educational Process Scheduling Lab
pub struct EducationalScheduler {
    current_process: Option<Process>,
    time_slice: Duration,
    learning_mode: bool,
}

impl EducationalScheduler {
    // Students implement different scheduling algorithms
    pub fn schedule_next(&mut self) -> Result<()> {
        if self.learning_mode {
            self.explain_decision()?;
        }
        self.select_next_process()
    }
}
